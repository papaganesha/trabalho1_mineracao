CREATE VIEW EXTRACT_CLIENTS AS SELECT A.ID_CLIENTE, A.ENDERECO FROM CLIENTES A, TABELA_CHAVE B WHERE A.ID_CLIENTE = B.ID AND B.TABELA = 'CLIENTES' AND B.CARGA = 0;

CREATE VIEW EXTRACT_STORES AS SELECT A.ID_LOJA, A.ENDERECO FROM LOJAS A, TABELA_CHAVE B WHERE A.ID_LOJA = B.ID AND B.TABELA = 'LOJAS' AND B.CARGA = 0

CREATE VIEW EXTRACT_BOOKS AS SELECT A.ID_LIVRO, A.TITULO FROM LIVROS A, TABELA_CHAVE B WHERE A.ID_LIVRO = B.ID AND B.TABELA = 'LIVROS' AND B.CARGA = 0

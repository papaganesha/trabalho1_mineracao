SOMANDO VALOR_COMPRA E VALOR_VENDA PARA DESCOBRIR FATURAMENTO LIQUIDO DE LOJAS

SELECT SUM(A.VALOR_VENDA) AS 'FATURAMENTO_BRUTO', SUM(A.VALOR_COMPRA) AS 'GASTOS', SUM(A.valor_venda) - SUM(a.valor_compra) as 'FATURAMENTO_LIQUIDO' FROM LIVROS A, ITENS_VENDAS B, VENDAS C WHERE B.id_livro = a.id_livro and B.id_venda = C.id_venda AND c.ID_LOJA = 7
